// This loop watches the game state 10 times a second and forces the colors setInterval(() => { const isInverted = document.body.classList.contains('inverted'); const gameCanvas = document.querySelector('.runner-canvas'); if (isInverted) { // Paint the background dark document.body.style.setProperty('background-color', '#202124', 'important'); // Flip the game canvas colors if (gameCanvas) { gameCanvas.style.setProperty('filter', 'invert(1)', 'important'); } } else { // Reset to day mode document.body.style.backgroundColor = ''; if (gameCanvas) { gameCanvas.style.filter = ''; } } }, 100);
